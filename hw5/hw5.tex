\section{โจทย์ : ถ้า $n$ , $m$ เป็น perfect square แล้ว $nm$ เป็น perfect square}
}
\begin{align}
	let \ n &= a^{2}, a \in \mathbb{Z} \\
	let \ m &= b^{2}, b \in \mathbb{Z} \\
\end{align}
\subsection{Proof by Direct Proof}
\begin{align}
	nm &= a^{2} \cdot b{2} \\
	nm &= (ab)^{2} \\
	let \ p &= ab \\
	nm &= p^{2}, p \in \mathbb{Z}
\end{align}
$\therefore$ ถ้า $n$ , $m$ เป็น perfect square แล้ว $nm$ เป็น perfect square
