{\small
\section{ให้ $f(x) = 3x - 2$ และ $g(x) = x^2 + 1$ จงหา $(g \circ f)(x)$ และ $(f \circ g)(x)$}
}
\subsection{$(g \circ f)(x)$}
\begin{align}
	(g \circ f)(x) &= f(x)^2 + 1 \\
	(g \circ f)(x) &= (3x - 2)^2 + 1
\end{align}

\subsection{$(f \circ g)(x)$}
\begin{align}
	(f \circ g)(x) &= 3(g(x)) - 2 \\
	(f \circ g)(x) &= 3(x^2 + 1) - 2 \\
	(f \circ g)(x) &= 3x^2 + 3 - 2 \\
	(f \circ g)(x) &= 3x^2 + 1
\end{align}

{\small
\section{จงหา $f^{-1}$ เมื่อ $f(x) = \frac{2x + 5}{3}$}
}
\subsection{$f^{-1}$}
\begin{align}
	let \quad y &= f(x) \\
	y &= \frac{2x + 5}{3} \\
	inverse ; \quad x &= \frac{2y + 5}{3} \\
	3x &= 2y + 5 \\
	3x - 5 &= 2y \\
	\frac{3x - 5}{2} &= y \\
	f^{-1} &= \frac{3x - 5}{2}
\end{align}
