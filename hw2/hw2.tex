\section{จงหา Domain และ Range ของ $f(x) = \frac{1}{3 - \sqrt{ x^2-25}}$}
}
\subsection{Domain}
\subsubsection{Cannot divide by 0}
\begin{align}
	3 - \sqrt{x^2-25} &\neq 0 \\
	-\sqrt{x^2-25} &\neq -3 \quad \text{(minus 3 both sides)} \\
	\sqrt{x^2-25} &\neq 3 \quad \text{(multiply -1 both sides)} \\
	x^2 - 25 &\neq 9 \quad \text{(square both sides)} \\
	x^2 - 34 &\neq 0 \quad \text{(minus 9 to both sides)} \\
	(x - \sqrt{34})(x + \sqrt{34}) &\neq 0 \\
	x &\neq \pm \sqrt{34} \\
	x &\in (-\infty, -\sqrt{34}) \cup (-\sqrt{34}, \sqrt{34}) \cup (\sqrt{34}, \infty)
\end{align}

\subsubsection{In square root cannot be less than 0}
\begin{align}
	x^2 - 25 &\geq 0 \\
	(x - 5)(x + 5) &\geq 0 \\
	x &\in (-\infty, -5] \cup [5, \infty)
\end{align}

\subsubsection{Combine}
\begin{align}
	x &\in ((-\infty, -\sqrt{34}) \cup (-\sqrt{34}, \sqrt{34}) \cup (\sqrt{34}, \infty)) \cap ((-\infty, -5] \cup [5, \infty)) \\
	\therefore x &\in (-\infty, -\sqrt{34}) \cup (-\sqrt{34}, -5] \cup [5, \sqrt{34}) \cup (\sqrt{34}, \infty)
\end{align}

\subsection{Range}
\begin{align}
	let \ t = \sqrt{x^2 - 25} ;\quad t \geq 0, t \neq 3
\end{align}

\subsubsection{t \rightarrow \ [0, 3)}
\begin{align}
	when \ t = 0;\quad y &= \frac{1}{3} \\
	when \ t \rightarrow \ 3^-;\quad y &\rightarrow \infty \\ 
	y &\in [1/3, \infty)
\end{align}

\subsubsection{t \rightarrow \ (3, $\infty$)}
\begin{align}
	when \ t \rightarrow \ 3^+;\quad y &\rightarrow -\infty \\ 
	when \ t \rightarrow \ \infty;\quad y &\rightarrow 0 \\
	y &\in (-\infty, 0)
\end{align}

\subsubsection{Combine}
\begin{align}
	\therefore y &\in (-\infty, 0) \cup [1/3, \infty)
\end{align}
